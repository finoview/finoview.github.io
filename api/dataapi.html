# 繁微指标数据API接口

# 繁微指标数据调用地址

https://finoview.com.cn/autoApi/IndexData

# 输入用户key、密码和参数

需要以post方式输入以下用户key、密码和参数

字段 字段类型 字段说明
superIndexLabel varchar 这里输入繁微指标目录结果中的指标编码
StartDate varchar(8) 开始日期
EndDate varchar(8) 结束日期
appkey 用户key
appsecret 用户key的密码

# 输出结果表字段

字段 字段类型 字段说明
f_id varchar 指标编码
f_name varchar 指标名称
f_resource varchar 数据来源
f_frequency varchar 频率
f_date varchar 日期
f_value varchar 数据
f_unit varchar 单位
f_explain varchar 备注

# python样例代码

import requests

url = "https://finoview.com.cn/autoApi/IndexData"

querystring = {"superIndexLabel":"DZ00012732","StartDate":"20190401","EndDate":"20190419","appkey":"请输入APPKEY","appsecret":"请输入APPSECRET"}

requests.packages.urllib3.disable_warnings()
response = requests.request("POST", url, params=querystring, verify=False)

print(response.text)
